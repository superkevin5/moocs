# http://calebmadrigal.com/display-list-as-table-in-ipython-notebook/
class ListTable(list):
    """ Overridden list class which takes a 2-dimensional list of 
        the form [[1,2,3],[4,5,6]], and renders an HTML Table in 
        IPython Notebook. """
    
    def _repr_html_(self):
        html = ["<table>"]
        for row in self:
            html.append("<tr>")
            
            for col in row:
                html.append("<td>{0}</td>".format(col))
            
            html.append("</tr>")
        html.append("</table>")
        return ''.join(html)

def merge_two_dicts(x, y):
    '''Given two dicts, merge them into a new dict as a shallow copy.'''
    z = x.copy()
    z.update(y)
    return z

def plot_question(data):
    option_scores = data['option_scores']
    yvals = data['yvals']
    title = data['title']
    option_texts = data['option_texts'] 
    question = data['question']
    question_type = data['question_type']
    option_numbers = data['option_numbers']
    yvals_percentages = data['yvals_percentages']
    num_distinct_students = data['num_distinct_students']
    
    
    number_of_options = len(option_texts)
    
    # values ordered by 'option_number'
    # the value here is the 'number of options' for the question (it defines printed ordering)
    xvals  = range(number_of_options)

    # labels ordered by 'option_number'
    xnames = ["(A{0})".format(x) for x in option_numbers]
 
    # option_score (1 or 0) - indicates green or red coloring (again ordered by 'option_number')
    # 0 = red
    # 1 = green
    option_scores = option_scores

    # the frequency counts of the items selected
    yvals = yvals

    # visually easier when bars have some interspacing
    width  = 0.5 # width of the bar

    # make the grid marker lines spread out into 5 equal sections.
    yinterval = max(yvals)/5 
    yinterval = max(yinterval, 1) # guard against 0

    # make the width of the figure based on the number of columns
    # keep height the same for all
    figure = plt.figure(figsize=(number_of_options*1.8,5))

    ax = plt.axes()
    ax.yaxis.grid(True, zorder=0)

    for x,y, y_percentage in zip(xvals, yvals, yvals_percentages):
        ax.text(x+(width-0.5)/2., 
                y, 
                'n=%1d \n%1.1f%%'%(y,y_percentage),
                va='bottom', 
                ha='center') 
    
    plt.title(title)
    plt.xlabel('Selected Answers')
    plt.ylabel('Frequency Count of Selected Answers')

    barlist = plt.bar(xvals, yvals, width=width, align='center', linewidth=0.5, zorder=3, color='lightblue', edgecolor='darkblue')

    # set the colour of the bar to indicate if it was the correct option or not.
    for index, value in enumerate(option_scores):
        if value == 0:
            color     = 'mistyrose'
            edgecolor = 'darkred'
        
        if value == 1:
            color     = 'lightgreen'
            edgecolor = 'darkgreen'

        barlist[index].set_color(color)
        barlist[index].set_edgecolor(edgecolor)


    plt.xticks(xvals, xnames)

    # force the xrange up 1 block, to give room for the label (the 2*yinterval part)
    plt.yticks(range(0,max(yvals)+2*yinterval,yinterval))

    # shift the x-range, so the column isn't flush on the left
    plt.xlim([min(xvals) - 0.5, max(xvals) + 0.5]) 

    # display the graph
    plt.show()

    print "Total selections made: {0} (from {1} distinct students)".format(sum(yvals), num_distinct_students)

# return a dataframe filtered to the lecture_id and question_number (i.e. return the possible answer choices)
# return also the order_number of the filter (since a order_number is not PK we couldn't pick it up earlier)
def get_question_metadata(df_lec_qs_with_options, lecture_id, question_number):
    df_lec_qs_with_options = df_lec_qs_with_options[(df_lec_qs_with_options['lecture_id']==lecture_id) & 
                                                    (df_lec_qs_with_options['question_number']==question_number)]

    df_lec_qs_with_options = df_lec_qs_with_options.sort(columns='option_number')
    
    question = None
    order_number = None
    option_texts = []
    option_numbers = []
    option_scores = []

    for _, row in df_lec_qs_with_options.iterrows():
        question = row['assessment_question_prompt']
        order_number = row['order_number']
        option_texts.append(row['option_text'])
        option_numbers.append(row['option_number'])
        option_scores.append(row['option_score'])
        
    # pandas makes None into Nans (revert to None to work with other routines).
    if pd.isnull(order_number):
        order_number = None
    
    data = {}
    data['question'] = question
    data['order_number'] = order_number
    data['option_texts'] = option_texts
    data['option_numbers'] = option_numbers
    data['option_scores'] = option_scores
    
    return data

# Helper function for the filters.
def get_plot_variables(question_metadata, df_payload, signature_track=None, first_attempt=None, last_attempt=None):
    _option_texts   = question_metadata['option_texts']
    _option_numbers = question_metadata['option_numbers']
    _option_scores  = question_metadata['option_scores']

    df_payload_lec_qs = df_payload[(df_payload['lecture_id']==lecture_id) &
                                   (df_payload['question_number']==question_number)]

    if signature_track is True:
        df_payload_lec_qs = df_payload_lec_qs[(df_payload_lec_qs['in_signature_track']==1)]

    if first_attempt is True:
        df_payload_lec_qs = df_payload_lec_qs[(df_payload_lec_qs['attempt_first']==1)]

    # last attempt as strict last attempt, so exclude if last attempt is also first attempt
    # i.e. exclude the cases, when attempt_max just equals 1
    if last_attempt is True:
        df_payload_lec_qs = df_payload_lec_qs[(df_payload_lec_qs['attempt_last']==1) & 
                                              (df_payload_lec_qs['attempt_first']!=1) ]
                                             
    option_scores = []
    yvals = []
    option_texts = []
    option_numbers = []
    yvals_percentages = []
  
    num_distinct_students = 0
    
    for index, option_text in enumerate(_option_texts):
        option_number = _option_numbers[index]
        option_score  = _option_scores[index]
    
        # these will all be the same, for each filter - to get the distinct count, we need to count the number of cells
        num_distinct_students = df_payload_lec_qs[(df_payload_lec_qs['option_number']==option_number)]['session_user_id'].drop_duplicates().count()

        frequency = df_payload_lec_qs[(df_payload_lec_qs['option_number']==option_number)]['option_selected'].sum()

        option_scores.append(option_score)
        yvals.append(frequency)
        option_texts.append(option_text)
        option_numbers.append(option_number)

    total = sum(yvals)
    for x in yvals:
        if total == 0:
            yvals_percentages.append(0)
        else:
            percentage = round(float(x)/float(total)*100.0,2)
            yvals_percentages.append(percentage)
        
        
    data = {}
    data['option_scores']     = option_scores
    data['yvals']             = yvals
    data['option_texts']      = option_texts
    data['option_numbers']    = option_numbers
    data['yvals_percentages'] = yvals_percentages
    data['num_distinct_students'] = num_distinct_students
    
    return data

def create_table(data):
    option_scores     = data['option_scores']
    yvals             = data['yvals']
    option_texts      = data['option_texts']
    option_numbers    = data['option_numbers']
    yvals_percentages = data['yvals_percentages']

    table = ListTable()
    x = ["A{0}".format(x) for x in data['option_numbers']]
    table.append(x)
    
    y = ["n={0}".format(x) for x in data['yvals']]
    table.append(y)

    z = ["{0}%".format(x) for x in data['yvals_percentages']]
    table.append(z)

    return table
